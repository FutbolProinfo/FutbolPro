            if (window.scrollY > 50) {
                nav.classList.add('bg-black');
                nav.classList.remove('bg-opacity-80');
            } else {
                nav.classList.add('bg-opacity-80');
            }
        });

        // Control de vídeos
        const videos = document.querySelectorAll('video');
        videos.forEach(video => {
            const container = video.parentElement;
            const playBtn = container.querySelector('button');

            if (playBtn) {
                playBtn.addEventListener('click', () => {
                    if (video.paused) {
                        video.play();
                        container.querySelector('.absolute').classList.add('hidden');
                    } else {
                        video.pause();
                        container.querySelector('.absolute').classList.remove('hidden');
                    }
                });
